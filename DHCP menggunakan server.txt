7.Router core
enable
configure terminal
hostname R-CORE

interface g0/0
 no shutdown

! VLAN 10 - SERVER (Gunakan IP .2)
interface g0/0.10
 encapsulation dot1Q 10
 ip address 192.168.10.2 255.255.255.0
 standby 10 ip 192.168.10.1
 standby 10 priority 110
 standby 10 preempt

! VLAN 20 - KEUANGAN
interface g0/0.20
 encapsulation dot1Q 20
 ip address 192.168.20.2 255.255.255.0
 ip helper-address 192.168.10.10
 standby 20 ip 192.168.20.1
 standby 20 priority 110
 standby 20 preempt
 ip access-group 101 out

! VLAN 30 - LAB KOMPUTER
interface g0/0.30
 encapsulation dot1Q 30
 ip address 192.168.30.2 255.255.255.0
 ip helper-address 192.168.10.10
 standby 30 ip 192.168.30.1
 standby 30 priority 110
 standby 30 preempt

interface g0/0.40
 encapsulation dot1Q 40
 ip address 192.168.40.2 255.255.255.0
 ip helper-address 192.168.10.10
 standby 40 ip 192.168.40.1
 standby 40 priority 110
 standby 40 preempt

! VLAN 50 - DOSEN
interface g0/0.50
 encapsulation dot1Q 50
 ip address 192.168.50.2 255.255.255.0
 ip helper-address 192.168.10.10
 standby 50 ip 192.168.50.1
 standby 50 priority 110
 standby 50 preempt

! VLAN 60 - RUANG KELAS
interface g0/0.60
 encapsulation dot1Q 60
 ip address 192.168.60.2 255.255.255.0
 ip helper-address 192.168.10.10
 standby 60 ip 192.168.60.1
 standby 60 priority 110
 standby 60 preempt

! SETTING ACL (KEAMANAN KEUANGAN)
access-list 101 permit ip 192.168.50.0 0.0.0.255 192.168.20.0 0.0.0.255
access-list 101 permit ip 192.168.10.0 0.0.0.255 192.168.20.0 0.0.0.255
access-list 101 deny ip any 192.168.20.0 0.0.0.255
access-list 101 permit ip any any
